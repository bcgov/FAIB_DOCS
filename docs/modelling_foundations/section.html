set.seed(999) 
generateFullSudoku <- function() {
  base <- matrix(0, 9, 9)
  
  # Simple valid base pattern
  for (i in 1:9) {
    for (j in 1:9) {
      base[i,j] <- ((i-1)*3 + floor((i-1)/3) + j - 1) %% 9 + 1
    }
  }
  
  # Randomly shuffle rows within blocks
  for (b in 0:2) {
    rows <- sample(1:3) + b*3
    base[(b*3+1):(b*3+3), ] <- base[rows, ]
  }
  
  # Randomly shuffle columns within blocks
  for (b in 0:2) {
    cols <- sample(1:3) + b*3
    base[, (b*3+1):(b*3+3)] <- base[, cols]
  }
  
  # Shuffle numbers 1–9 globally
  perm <- sample(1:9)
  base <- matrix(perm[base], 9, 9)
  
  return(base)
}

createPuzzle <- function(fullSudoku, emptyFraction = 0.6) {
  puzzle <- fullSudoku
  
  # Total cells to remove
  n_remove <- round(81 * emptyFraction)
  
  # Random positions
  positions <- sample(1:81, n_remove)
  
  puzzle[positions] <- 0
  
  return(puzzle)
}


generateSudokuWithZeros <- function(emptyFraction = 0.6) {
  full <- generateFullSudoku()
  puzzle <- createPuzzle(full, emptyFraction)
  return(puzzle)
}

printSudoku <- function(sudoku) {
  for (i in 1:9) {
    # Print horizontal divider between 3x3 blocks
    if (i %% 3 == 1 && i != 1) cat("---------------------\n")
    
    line <- ""
    for (j in 1:9) {
      # Print vertical divider
      if (j %% 3 == 1 && j != 1) line <- paste0(line, "| ")
      line <- paste0(line, sudoku[i,j], " ")
    }
    cat(line, "\n")
  }
  cat("\n")
}

estimateInitialSigma <- function(sudoku, fixed, blocks) {
  samples <- c()
  tmp <- sudoku
  
  for (i in 1:20) {
    block <- blocks[[sample(1:9,1)]]
    tmp <- swapCells(tmp, fixed, block)
    samples <- c(samples, calcErrors(tmp))
  }
  
  return(sd(samples))
}


# ==========================================================
# COST FUNCTION (OBJECTIVE FUNCTION)

# The following calculates the total number of rule violations in 
# the Sudoku grid by summing 
# how many duplicate values exist across all rows and columns.
# ==========================================================
# This measures how "wrong" the Sudoku is.
#
# Idea:
# - Each row and column should contain numbers 1–9 exactly once.
# - If duplicates exist, we count errors.
#
# Lower cost = better solution
# Perfect solution = cost = 0

calcErrors <- function(sudoku) {
  errors <- 0
  for (i in 1:9) {
    # Count duplicates in row i and column i
    errors <- errors +
      (9 - length(unique(sudoku[i,]))) +   # row errors
      (9 - length(unique(sudoku[,i])))     # column errors
  }
  return(errors)
}


#Define 3 X3 Blocks
# ==========================================================
# Returns list of 9 blocks, each containing coordinates

getBlocks <- function() {
  blocks <- list()
  idx <- 1
  for (br in 0:2) {
    for (bc in 0:2) {
      cells <- expand.grid(
        (1:3) + br*3,
        (1:3) + bc*3
      )
      blocks[[idx]] <- as.matrix(cells)
      idx <- idx + 1
    }
  }
  return(blocks)
}


# ==========================================================
# 6. INITIAL RANDOM SOLUTION
# ==========================================================
# Fill each 3x3 block with missing numbers randomly
#
# IMPORTANT DESIGN CHOICE:
# - Each 3x3 block is always VALID (contains numbers 1–9)
# - Only rows and columns may contain conflicts

fillBlocks <- function(sudoku, blocks) {
  for (block in blocks) {
    values <- sudoku[block]
    
    # Find numbers missing in this block
    missing <- setdiff(1:9, values)
    
    # Fill empty cells randomly with missing numbers
    zeros <- which(values == 0)
    values[zeros] <- sample(missing)
    
    sudoku[block] <- values
  }
  return(sudoku)
}

# ==========================================================
# GENERATE A NEIGHBOR (STATE TRANSITION)

# We create a "new candidate solution" by:
# - Picking a random 3x3 block
# - Swapping two NON-FIXED cells inside it
#
# This ensures:
# - Block remains valid
# - Only row/column structure changes

swapCells <- function(sudoku, fixed, block) {
  repeat {
    idx <- sample(1:nrow(block), 2)
    c1 <- block[idx[1],]
    c2 <- block[idx[2],]
    
    # Only swap if both are NOT fixed
    if (fixed[c1[1],c1[2]] == 0 &&
        fixed[c2[1],c2[2]] == 0) {
      
      # Swap values
      temp <- sudoku[c1[1],c1[2]]
      sudoku[c1[1],c1[2]] <- sudoku[c2[1],c2[2]]
      sudoku[c2[1],c2[2]] <- temp
      
      return(sudoku)
    }
  }
}

3.0.1 Simulated Annealing Solver

The solveSudoku function applies a simulated annealing optimization process to solve a Sudoku puzzle by iteratively reducing conflicts in rows and columns while preserving valid 3×3 blocks (Constraint Decomposition). It begins by identifying fixed cells, generating an initial valid state by randomly filling each block, and estimating an initial temperature (sigma) that controls how freely the algorithm explores different configurations. During each iteration, it proposes a new candidate solution by swapping two non-fixed cells within a random block, evaluates whether the change improves the error (number of conflicts), and accepts it either if it is better or with a probability that allows occasional acceptance of worse solutions to escape local minima. As the algorithm progresses, the temperature gradually decreases (cooling), making the search increasingly focused and less random, while continuously tracking the best solution found. The process continues until either a conflict-free Sudoku (error = 0) is reached or a maximum number of iterations is exceeded, at which point the best available solution is returned.
